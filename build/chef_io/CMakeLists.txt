cmake_minimum_required(VERSION 2.6)

SET(PROJECTNAME chefio)
SET(LIBRARY_OUTPUT_PATH ${ROOT_DIR}/lib)

FILE(GLOB_RECURSE SOURCES "${ROOT_DIR}/chef_io/*.cc")
ADD_DEFINITIONS("-g -Wall -Werror -fPIC")
PROJECT(${PROJECTNAME})

INCLUDE_DIRECTORIES(${ROOT_DIR}/chef_base)
LINK_DIRECTORIES(${ROOT_DIR}/lib)
ADD_LIBRARY(${PROJECTNAME} SHARED ${SOURCES})
TARGET_LINK_LIBRARIES(${PROJECTNAME} chefbase)
